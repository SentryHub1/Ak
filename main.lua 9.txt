function CorrectKeyNotifications()

        Name = "Correct Key",
        Content = "Welcome to Sky Arc",

end

function WrongKeyNotifications()

    Name = "Enter Key",
    Default = "Enter Key",
    TextDisappear = true,
    Callback = function(Value)
